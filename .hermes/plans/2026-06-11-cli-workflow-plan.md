# DataGangeR CLI Workflow Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a CRAN-safe terminal CLI for profiling data, detecting roles, generating synthesis specs, synthesizing/exporting bundles, and inspecting exported bundles.

**Architecture:** Put a tiny executable shim at `exec/dataganger` and keep all logic in `R/cli.R` so it is testable through normal package tests. Use a custom lightweight argument parser built on `commandArgs()` semantics, `jsonlite` for JSON, `yaml` for YAML, and existing DataGangeR engine functions for all data work. CLI functions return integer status codes for tests; only the shim calls `quit()`.

**Tech Stack:** R package code, `cli`, `jsonlite`, `yaml`, `readr`, `utils`, `testthat`, existing `read_input()`, `profile_data()`, `detect_roles()`, `synth_spec()`, `synthesize_data()`, `compare_synthetic()`, `privacy_check()`, and `export_synthetic()`.

## Command-to-Engine Mapping

| CLI command | Engine calls | Output | Clean status |
|---|---|---|---|
| `dataganger profile real_data.sas7bdat --out profile.json` | `read_input(path)`, `profile_data(data)`, `jsonlite::write_json()` | JSON profile with `n_rows`, `n_cols`, `generated_at`, `profile` rows | `0` |
| `dataganger roles real_data.sas7bdat --out roles.yaml` | `read_input(path)`, `profile_data(data)`, `detect_roles(data, profile)` , `yaml::write_yaml()` | YAML list containing `roles` rows | `0` |
| `dataganger spec --purpose ai_programming --out spec.yaml` | `synth_spec(purpose = purpose)` , `yaml::write_yaml()` | YAML synthesis spec | `0` |
| `dataganger synthesize real_data.sas7bdat --spec spec.yaml --out synthetic_bundle.zip` | `read_input(path)`, `yaml::read_yaml()`, `detect_roles()`, `synth_spec()` reconstruction, `privacy_check(stage = "pre")`, `synthesize_data()`, `compare_synthetic()`, `privacy_check(stage = "post")`, `export_synthetic(format = "zip")` | Zip bundle | `0` |
| `dataganger inspect synthetic_bundle.zip` | `utils::unzip(list = TRUE)`, `jsonlite::read_json(manifest.json)`, `readr::read_csv(data_dictionary.csv)`, `readLines(privacy_report.txt)` | Console summary only; no original data required | `0` |

## Status Codes

- `0`: successful command or help output.
- `1`: processing error from file IO, package engine validation, serialization, or bundle inspection.
- `2`: syntax error, missing required argument, unknown command, unknown option, or invalid status-independent CLI usage.

```r
#' DataGangeR command-line interface
#'
#' Testable command-line entrypoint used by the installed `exec/dataganger` shim.
#'
#' @param args Character vector of trailing command-line arguments.
#' @param quit Logical. When `TRUE`, terminate the R process using the returned
#'   status code. Tests pass `FALSE` and assert the integer code.
#'
#' @return Integer status code: `0` success, `1` processing error, `2` syntax error.
#' @export
dataganger_cli <- function(args = commandArgs(trailingOnly = TRUE), quit = TRUE) {
  status <- cli_dispatch(args)
  if (isTRUE(quit)) {
    base::quit(save = "no", status = status, runLast = FALSE)
  }
  status
}

cli_status_ok <- function() 0L
cli_status_error <- function() 1L
cli_status_usage <- function() 2L

cli_dispatch <- function(args) {
  tryCatch(
    {
      if (length(args) == 0L || identical(args[[1]], "--help") || identical(args[[1]], "-h")) {
        cli_print_help()
        return(cli_status_ok())
      }

      command <- args[[1]]
      rest <- args[-1]

      switch(
        command,
        profile = cli_cmd_profile(rest),
        roles = cli_cmd_roles(rest),
        spec = cli_cmd_spec(rest),
        synthesize = cli_cmd_synthesize(rest),
        inspect = cli_cmd_inspect(rest),
        {
          cli::cli_alert_danger("Unknown command: {command}")
          cli_status_usage()
        }
      )
    },
    dataganger_cli_usage_error = function(e) {
      cli::cli_alert_danger(conditionMessage(e))
      cli_status_usage()
    },
    error = function(e) {
      cli::cli_alert_danger(conditionMessage(e))
      cli_status_error()
    }
  )
}

cli_print_help <- function() {
  cat(
    paste(
      "Usage: dataganger <command> [options]",
      "",
      "Commands:",
      "  profile <data-file> --out <profile.json>",
      "  roles <data-file> --out <roles.yaml>",
      "  spec --purpose <purpose> --out <spec.yaml>",
      "  synthesize <data-file> --spec <spec.yaml> --out <synthetic_bundle.zip>",
      "  inspect <synthetic_bundle.zip>",
      sep = "\n"
    ),
    "\n",
    sep = ""
  )
}

cli_usage_error <- function(message) {
  structure(
    list(message = message, call = NULL),
    class = c("dataganger_cli_usage_error", "error", "condition")
  )
}

cli_cmd_profile <- function(args) {
  cli_usage_error("profile requires exactly one data file")
}

cli_cmd_roles <- function(args) {
  cli_usage_error("roles requires exactly one data file")
}

cli_cmd_spec <- function(args) {
  cli_usage_error("Missing required option --purpose")
}

cli_cmd_synthesize <- function(args) {
  cli_usage_error("synthesize requires exactly one data file")
}

cli_cmd_inspect <- function(args) {
  cli_usage_error("inspect requires exactly one bundle file")
}
```

```r
cli_parse_options <- function(args, allowed = character()) {
  positionals <- character()
  options <- list()
  i <- 1L

  while (i <= length(args)) {
    token <- args[[i]]
    if (startsWith(token, "--")) {
      key <- substring(token, 3L)
      if (!(key %in% allowed)) {
        stop(cli_usage_error(sprintf("Unknown option --%s", key)))
      }
      if (i == length(args) || startsWith(args[[i + 1L]], "--")) {
        stop(cli_usage_error(sprintf("Missing value for option --%s", key)))
      }
      options[[key]] <- args[[i + 1L]]
      i <- i + 2L
    } else {
      positionals <- c(positionals, token)
      i <- i + 1L
    }
  }

  list(positionals = positionals, options = options)
}

cli_require_option <- function(parsed, name) {
  value <- parsed$options[[name]]
  if (is.null(value) || !nzchar(value)) {
    stop(cli_usage_error(sprintf("Missing required option --%s", name)))
  }
  value
}

cli_require_n_positionals <- function(parsed, n, command, label) {
  if (length(parsed$positionals) != n) {
    stop(cli_usage_error(sprintf("%s requires exactly %d %s", command, n, label)))
  }
  parsed$positionals
}

cli_assert_existing_file <- function(path) {
  if (!file.exists(path)) {
    stop(sprintf("Input file does not exist: %s", path), call. = FALSE)
  }
  invisible(path)
}
```

```r
cli_unpack_bundle_file <- function(zip_path, member, tmp) {
  utils::unzip(zip_path, files = member, exdir = tmp, junkpaths = TRUE)

}

cli_read_bundle_summary <- function(zip_path) {
  cli_assert_existing_file(zip_path)
  listing <- utils::unzip(zip_path, list = TRUE)
  required <- c("manifest.json", "data_dictionary.csv", "privacy_report.txt")
  missing <- setdiff(required, listing$Name)
  if (length(missing) > 0L) {
    stop(sprintf("Bundle is missing required file: %s", paste(missing, collapse = ", ")), call. = FALSE)
  }

  tmp <- tempfile("dataganger-inspect-")
  dir.create(tmp)
  on.exit(unlink(tmp, recursive = TRUE, force = TRUE), add = TRUE)

  manifest_path <- cli_unpack_bundle_file(zip_path, "manifest.json", tmp)
  dictionary_path <- cli_unpack_bundle_file(zip_path, "data_dictionary.csv", tmp)
  privacy_path <- cli_unpack_bundle_file(zip_path, "privacy_report.txt", tmp)

  manifest <- jsonlite::read_json(manifest_path, simplifyVector = TRUE)
  dictionary <- readr::read_csv(dictionary_path, show_col_types = FALSE)
  privacy <- readLines(privacy_path, warn = FALSE)

  list(
    manifest = manifest,
    dictionary = dictionary,
    privacy = privacy,
    files = listing$Name
  )
}

cli_print_bundle_summary <- function(summary) {
  manifest <- summary$manifest
  dictionary <- summary$dictionary
  privacy <- summary$privacy

  cat("Synthetic bundle\n")
  cat(sprintf("Purpose: %s\n", manifest$purpose %||% "unknown"))
  cat(sprintf("Variables: %d\n", nrow(dictionary)))
  cat(sprintf("Files: %d\n", length(summary$files)))

  if ("type" %in% names(dictionary)) {
    type_counts <- sort(table(dictionary$type), decreasing = TRUE)
    cat("Schema types:\n")
    for (nm in names(type_counts)) {
      cat(sprintf("  - %s: %d\n", nm, unname(type_counts[[nm]])))
    }
  }

  privacy_lines <- privacy[nzchar(privacy)]
  cat("Privacy exposure ratings:\n")
  if (length(privacy_lines) == 0L) {
    cat("  - No privacy report lines found\n")
  } else {
    for (line in head(privacy_lines, 8L)) {
      cat(sprintf("  - %s\n", line))
    }
  }
}
```
